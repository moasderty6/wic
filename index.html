$w.onReady(function () { // عند تغيير القيم $w('#numRooms').onInput(updatePrice); $w('#numNights').onInput(updatePrice); function updatePrice() { const rooms = Number($w('#numRooms').value); const nights = Number($w('#numNights').value); // تحقق من أن القيم أرقام صالحة if (!rooms || !nights || rooms < 0 || nights < 0) { $w('#totalPrice').text = "—"; return; } const roomPrice = 600; const total = rooms * nights * roomPrice; $w('#totalPrice').text = `${total} AED`; } });
