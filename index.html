$w.onReady(function () { // حدث عند تغيير اختيار الغرف أو الأيام $w('#numRooms').onChange(updatePrice); $w('#numNights').onChange(updatePrice); function updatePrice() { const rooms = Number($w('#numRooms').value); const nights = Number($w('#numNights').value); // التحقق من أن القيم موجودة وصحيحة if (!rooms || !nights || rooms < 0 || nights < 0) { $w('#totalPrice').text = "—"; return; } const roomPrice = 600; const total = rooms * nights * roomPrice; $w('#totalPrice').text = `${total} AED`; } // لتحديث السعر عند أول تحميل updatePrice(); });
